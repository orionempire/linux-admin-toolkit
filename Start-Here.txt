# Welcome to the linux admin toolkit 00.00.03

----------------------------Installation Instructions Redhat EL 6.3----------------------------------------
#Starting with the developer workstation build

#install mysql
yum install mysql-server
service mysqld start
chkconfig mysqld on

yum install MySQL-python

yum install mod_wsgi

#Install pip
curl http://python-distribute.org/distribute_setup.py | python

#Install Django and modules
pip install Django
pip install xlwt
pip install xlrd
pip install python-nmap

mkdir /opt/linux-admin-toolkit
mkdir /etc/linux-admin-toolkit/

wget http://linux-admin-toolkit.googlecode.com/files/linux-admin-toolkit_(Build Date).tgz
tar xvfz linux-admin-toolkit_(Build Date).tgz

cp -R * /opt/linux-admin-toolkit/
cp -R /opt/linux-admin-toolkit/local-config-example/* /etc/linux-admin-toolkit/

#edit /etc/linux-admin-toolkit/frontend/local_settings.py

mysql -u root  -e 'CREATE SCHEMA `linux-admin-toolkit`'
python manage.py syncdb --noinput
python manage.py createsuperuser --username sysadmin --email sysadmin@linux-admin-toolkit.com

# Change your secret key in /opt/linux-admin-toolkit/frontend/frontend/settings.py
# python -c 'import random; print "".join([random.choice("abcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*(-_=+)") for i in range(50)])'

#add to  /etc/httpd/conf/httpd.conf
<VirtualHost *:80>

    ServerName wsgi.djangoserver
    DocumentRoot /opt/linux-admin-toolkit/frontend

    <Directory /opt/linux-admin-toolkit/frontend>
        Order allow,deny
        Allow from all
    </Directory>

    WSGIScriptAlias / /opt/linux-admin-toolkit/frontend/frontend/wsgi.py

    Alias /static/admin/ /usr/lib/python2.6/site-packages/django/contrib/admin/static/admin/

    <Directory /usr/lib/python2.6/site-packages/django/contrib/admin/static/admin>
        Order deny,allow
        Allow from all
    </Directory>

</VirtualHost>

#turn on apache
service httpd start
chkconfig httpd on
